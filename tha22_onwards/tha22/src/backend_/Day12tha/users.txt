var express = require('express');
var router = express.Router();
const Pool = require('pg').Pool;

const pool = new Pool({
  user:'postgres',
  host:'localhost',
  database:'postgres',
  password: '1234@567',
  port:5432
})
/* GET users listing. */
router.get('/', function(req, res, next) {
  pool.query('SELECT * FROM "Users"',(err,result) => {
    if(err) {
      throw err;
    }
    res.status(300),json(result);
  })
});

module.exports = router;
